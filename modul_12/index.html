Validasi Email 
    <script>
        function validasiEmail(email){
            var pola = /^\w+@[a-zA-Z_]+?\.[a-zA-Z]{2,3}$/;
            if(pola.test(email) == true){
                alert("Email valid");
            }else{
                alert("Email tidak valid");
            }
        }
    </script>
    <input type="text" id="email">
    <button onclick="validasiEmail(document.getElementById('email').value)">
        Cek
    </button>

Alert,Confirm,Prompt 
    <script>
        function alert1(){
            alert("Ini adalah alert");
        }
        function confirm1(){
            var x = confirm("Apakah anda yakin?");
            if(x == true){
                alert("Anda menekan tombol OK");
            }else{
                alert("Anda menekan tombol Cancel");
            }
        }
        function prompt1(){
            var x = prompt("Masukkan nama anda");
            alert("Halo "+x);
        }
    </script>
    <button onclick="alert1()">
        Alert
    </button>
    <button onclick="confirm1()">
        Confirm
    </button>
    <button onclick="prompt1()">
        Prompt
    </button>
